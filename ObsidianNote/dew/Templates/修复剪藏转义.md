<%*
const content = tp.file.content
const lines = content.split("\n")
const result = []
let inCodeBlock = false
let fixCount = 0

function fixEscapes(str) {
    var s = str
    s = s.split("\\_").join("_")
    s = s.split("\\#").join("#")
    s = s.split("\\*").join("*")
    s = s.split("\\[").join("[")
    s = s.split("\\]").join("]")
    s = s.split("\\-").join("-")
    s = s.replace(/(\d+)\\\./g, "$1.")
    return s
}

for (var i = 0; i < lines.length; i++) {
    var trimmed = lines[i].trimStart()
    if (trimmed.indexOf("```") === 0 || trimmed.indexOf("~~~") === 0) {
        inCodeBlock = !inCodeBlock
        result.push(lines[i])
        continue
    }

    if (inCodeBlock) {
        var cleaned = fixEscapes(lines[i])
        if (cleaned !== lines[i]) fixCount++
        result.push(cleaned)
    } else {
        result.push(lines[i])
    }
}

var final = result.join("\n")
if (final !== content) {
    await app.vault.modify(tp.file.find_tfile(tp.file.path), final)
}
%>
